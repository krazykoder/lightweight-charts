Histogram Series Examples

Histogram with Squeeze Data

const chart = LightweightCharts.createChart(document.getElementById('chart-container1'), {
    width: 600,
    height: 300,
});
const histogramSeries = chart.addHistogramSeries();
const sqzColors = { 1: '#26a69a', 2: '#ef5350', 3: '#64b5f6', 4: '#ffc107' };
const randomSlice1 = getRandomSlice();
const histogramData = data.map(d => ({
    time: d.time,
    value: d.SQZ_Hist,
    color: sqzColors[d.SQZ_HistC] || 'gray',
})).slice(randomSlice1.start, randomSlice1.end);
histogramSeries.setData(histogramData);

Histogram with Custom Base

const chart2 = LightweightCharts.createChart(document.getElementById('chart-container2'), {
    width: 600,
    height: 300,
});
const histogramSeries2 = chart2.addHistogramSeries({
    base: 0,
});
const randomSlice2 = getRandomSlice();
const histogramData2 = data.map(d => ({
    time: d.time,
    value: d.SQZ_Hist,
})).slice(randomSlice2.start, randomSlice2.end);
histogramSeries2.setData(histogramData2);
