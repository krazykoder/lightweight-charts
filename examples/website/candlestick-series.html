Basic Candlestick Chart

    width: 600,
    height: 300,
});
const candlestickSeries = chart.addCandlestickSeries();

Customized Candlestick Chart

const chart2 = LightweightCharts.createChart(document.getElementById('chart-container2'), {
    width: 600,
    height: 300,
});
const candlestickSeries2 = chart2.addCandlestickSeries({
    upColor: '#26a69a',
    downColor: '#ef5350',
    borderDownColor: '#ef5350',
    borderUpColor: '#26a69a',
    wickDownColor: '#ef5350',
    wickUpColor: '#26a69a',
});
const randomSlice2 = getRandomSlice();
candlestickSeries2.setData(data.slice(randomSlice2.start, randomSlice2.end));

Candlesticks with Multiple EMAs

const chart3 = LightweightCharts.createChart(document.getElementById('chart-container3'), {
    width: 600,
    height: 300,
});
const candlestickSeries3 = chart3.addCandlestickSeries();
const randomSlice3 = getRandomSlice();
candlestickSeries3.setData(data.slice(randomSlice3.start, randomSlice3.end));

const ema21Series = chart3.addLineSeries({ color: 'blue', lineWidth: 1 });
ema21Series.setData(data.slice(randomSlice3.start, randomSlice3.end).map(d => ({ time: d.time, value: d.EMA_21 })));

const ema42Series = chart3.addLineSeries({ color: 'red', lineWidth: 1 });
ema42Series.setData(data.slice(randomSlice3.start, randomSlice3.end).map(d => ({ time: d.time, value: d.EMA_42 })));
