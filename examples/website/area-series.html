Area Chart with Gaps

const chart3 = LightweightCharts.createChart(document.getElementById('chart-container3'), {
    width: 600,
    height: 300,
});
const areaSeries3 = chart3.addAreaSeries({
    lineType: LightweightCharts.LineType.WithGaps,
});
const randomSlice3 = getRandomSlice();
const gappedData = data.map(d => ({ time: d.time, value: d.value })).slice(randomSlice3.start, randomSlice3.end);
gappedData[10].value = null;
gappedData[20].value = null;
areaSeries3.setData(gappedData);

Area Chart with Multiple EMAs

const chart4 = LightweightCharts.createChart(document.getElementById('chart-container4'), {
    width: 600,
    height: 300,
});
const areaSeries4 = chart4.addAreaSeries({
    topColor: 'rgba(33, 150, 243, 0.4)',
    bottomColor: 'rgba(33, 150, 243, 0)',
    lineColor: 'rgba(33, 150, 243, 1)',
});
const randomSlice4 = getRandomSlice();
areaSeries4.setData(data.map(d => ({ time: d.time, value: d.close })).slice(randomSlice4.start, randomSlice4.end));

const ema21Series = chart4.addLineSeries({ color: 'purple', lineWidth: 1 });
ema21Series.setData(data.slice(randomSlice4.start, randomSlice4.end).map(d => ({ time: d.time, value: d.EMA_21 })));

const ema42Series = chart4.addLineSeries({ color: 'orange', lineWidth: 1 });
ema42Series.setData(data.slice(randomSlice4.start, randomSlice4.end).map(d => ({ time: d.time, value: d.EMA_42 })));
