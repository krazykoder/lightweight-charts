Line with Gaps

const chart3 = LightweightCharts.createChart(document.getElementById('chart-container3'), {
    width: 600,
    height: 300,
});
const lineSeries3 = chart3.addLineSeries({
    lineType: LightweightCharts.LineType.WithGaps,
    color: '#E91E63',
});
const randomSlice3 = getRandomSlice();
const gappedData = data.map(d => ({ time: d.time, value: d.value })).slice(randomSlice3.start, randomSlice3.end);
// create some gaps
gappedData[10].value = null;
gappedData[25].value = null;
lineSeries3.setData(gappedData);

Customized Line

const chart4 = LightweightCharts.createChart(document.getElementById('chart-container4'), {
    width: 600,
    height: 300,
});
const lineSeries4 = chart4.addLineSeries({
    color: '#2196f3',
    lineWidth: 3,
    lineStyle: LightweightCharts.LineStyle.Dotted,
    crosshairMarkerVisible: true,
    crosshairMarkerRadius: 6,
    crosshairMarkerBorderColor: '#ffffff',
    crosshairMarkerBackgroundColor: '#2296f3',
});
const randomSlice4 = getRandomSlice();
lineSeries4.setData(data.map(d => ({ time: d.time, value: d.value })).slice(randomSlice4.start, randomSlice4.end));
