Squeeze Indicator Example

Candlestick Chart with Squeeze Indicator

// Main Chart (Candlesticks)
const chart = LightweightCharts.createChart(
    document.getElementById('chart-container1'), 
    { width: 800, height: 400, timeScale: { tickMarkFormatter } }
);
const candlestickSeries = chart.addCandlestickSeries();
const randomSlice = getRandomSlice();
candlestickSeries.setData(data.slice(randomSlice.start, randomSlice.end));

// Histogram Pane (Squeeze Indicator)
const histogramPane = LightweightCharts.createChart(
    document.getElementById('chart-container2'),
    { width: 800, height: 200, timeScale: { tickMarkFormatter } }
);
const histogramSeries = histogramPane.addHistogramSeries({ base: 0 });
const squeezeMarkersSeries = histogramPane.addLineSeries({
    color: 'transparent', // Hide the line
    lineWidth: 0,
    crosshairMarkerVisible: false,
    lastValueVisible: false,
    priceLineVisible: false,
});

const sqzColors = { 1: '#26a69a', 2: '#ef5350', 3: '#64b5f6', 4: '#ffc107' };
const histogramData = data.map(d => ({
    time: d.time,
    value: d.SQZ_Hist,
    color: sqzColors[d.SQZ_HistC] || 'gray',
})).slice(randomSlice.start, randomSlice.end);
histogramSeries.setData(histogramData);

const squeezeMarkerData = data.map(d => ({
    time: d.time,
    value: d.SQZ === 1 ? 0 : null, // Place marker at 0 if squeeze is on
})).slice(randomSlice.start, randomSlice.end);
squeezeMarkersSeries.setMarkers(squeezeMarkerData.filter(d => d.value === 0).map(d => ({
    time: d.time,
    position: 'inBar',
    color: 'yellow',
    shape: 'circle',
    size: 0.5,
})));

// Sync main chart and histogram pane
let isSyncing = false;
chart.timeScale().subscribeVisibleTimeRangeChange(range => {
    if (isSyncing) return;
    isSyncing = true;
    if (range) histogramPane.timeScale().setVisibleRange(range);
    isSyncing = false;
});
histogramPane.timeScale().subscribeVisibleTimeRangeChange(range => {
    if (isSyncing) return;
    isSyncing = true;
    if (range) chart.timeScale().setVisibleRange(range);
    isSyncing = false;
});
