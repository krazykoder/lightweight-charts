Histogram Area Series Examples

Histogram Area Style

const chart = LightweightCharts.createChart(document.getElementById('chart-container1'), {
    width: 600,
    height: 300,
});
const histogramSeries = chart.addHistogramSeries({
    // HistogramStyle.Area = 1
    histogramStyle: 1, 
});
const sqzColors = { 1: '#26a69a', 2: '#ef5350', 3: '#64b5f6', 4: '#ffc107' };
const randomSlice1 = getRandomSlice();
const histogramData = data.map(d => ({
    time: d.time,
    value: d.SQZ_Hist,
    color: sqzColors[d.SQZ_HistC] || 'gray',
})).slice(randomSlice1.start, randomSlice1.end);
histogramSeries.setData(histogramData);
