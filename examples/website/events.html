Events Examples

Click and Crosshair Move Events

const chart = LightweightCharts.createChart(document.getElementById('chart-container1'), { width: 600, height: 300 });
const lineSeries = chart.addLineSeries();
const randomSlice1 = getRandomSlice();
lineSeries.setData(data.map(d => ({ time: d.time, value: d.value })).slice(randomSlice1.start, randomSlice1.end));

chart.subscribeClick(param => {
    console.log(`Click at (${param.point.x}, ${param.point.y}), time: ${param.time}`);
});

chart.subscribeCrosshairMove(param => {
    console.log(`Crosshair at (${param.point.x}, ${param.point.y}), time: ${param.time}`);
});

Syncing Multiple Charts

const chart2 = LightweightCharts.createChart(document.getElementById('chart-container2'), { width: 600, height: 300 });
const areaSeries2 = chart2.addAreaSeries();
const randomSlice2 = getRandomSlice();
areaSeries2.setData(data.map(d => ({ time: d.time, value: d.close })).slice(randomSlice2.start, randomSlice2.end));

const chart3 = LightweightCharts.createChart(document.getElementById('chart-container3'), { width: 600, height: 300 });
const barSeries3 = chart3.addBarSeries();
const randomSlice3 = getRandomSlice();
barSeries3.setData(data.slice(randomSlice3.start, randomSlice3.end));

// Sync crosshairs
chart2.subscribeCrosshairMove(param => {
    if (param.point) chart3.moveCrosshair(param.point);
});
chart3.subscribeCrosshairMove(param => {
    if (param.point) chart2.moveCrosshair(param.point);
});

// Sync time scales
chart2.timeScale().subscribeVisibleTimeRangeChange(range => {
    chart3.timeScale().setVisibleRange(range);
});
chart3.timeScale().subscribeVisibleTimeRangeChange(range => {
    chart2.timeScale().setVisibleRange(range);
});
