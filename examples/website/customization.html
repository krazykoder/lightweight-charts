General Customization

const chart = LightweightCharts.createChart(document.getElementById('chart-container1'), {
    width: 600,
    height: 300,
    layout: {
        backgroundColor: '#FAEBD7',
        textColor: '#696969',
    },
    grid: {
        vertLines: { color: 'rgba(70, 130, 180, 0.5)' },
        horzLines: { color: 'rgba(70, 130, 180, 0.5)' },
    },
    priceScale: { borderColor: '#483D8B' },
    timeScale: { borderColor: '#483D8B' },
});
const lineSeries = chart.addLineSeries();
const randomSlice1 = getRandomSlice();
lineSeries.setData(data.map(d => ({ time: d.time, value: d.value })).slice(randomSlice1.start, randomSlice1.end));

Crosshair: Styled Lines

const chart2 = LightweightCharts.createChart(document.getElementById('chart-container2'), {
    width: 600,
    height: 300,
    crosshair: {
        vertLine: {
            color: '#FF0000',
            width: 2,
            style: LightweightCharts.LineStyle.Dashed,
            labelVisible: true,
        },
        horzLine: {
            color: '#0000FF',
            width: 2,
            style: LightweightCharts.LineStyle.Dotted,
            labelVisible: true,
        },
    },
});
const barSeries2 = chart2.addBarSeries();
const randomSlice2 = getRandomSlice();
barSeries2.setData(data.slice(randomSlice2.start, randomSlice2.end));

Crosshair: Magnet Mode

const chart3 = LightweightCharts.createChart(document.getElementById('chart-container3'), {

    height: 300,

});
const lineSeries3 = chart3.addLineSeries({ color: '#26a69a' });

Crosshair: Normal (Free-move)

const chart4 = LightweightCharts.createChart(document.getElementById('chart-container4'), {
    width: 600,
    height: 300,
    crosshair: {
        mode: LightweightCharts.CrosshairMode.Normal,
    },
});
const candlestickSeries4 = chart4.addCandlestickSeries({ upColor: '#26a69a', downColor: '#ef5350', borderDownColor: '#ef5350', borderUpColor: '#26a69a', wickDownColor: '#ef5350', wickUpColor: '#26a69a' });
const randomSlice4 = getRandomSlice();
candlestickSeries4.setData(data.slice(randomSlice4.start, randomSlice4.end));
