Series Markers Examples

Playbook: Interactive Markers

Markers on Line Series

const chart1 = LightweightCharts.createChart(document.getElementById('chart-container1'), { width: 600, height: 300 });
const lineSeries1 = chart1.addLineSeries();
const randomSlice1 = getRandomSlice();
const slicedData1 = data.map(d => ({ time: d.time, value: d.value })).slice(randomSlice1.start, randomSlice1.end);
lineSeries1.setData(slicedData1);
lineSeries1.setMarkers([
    { time: slicedData1[Math.floor(slicedData1.length * 0.2)].time, position: 'aboveBar', color: '#f68410', shape: 'circle', text: 'Signal 1' },
    { time: slicedData1[Math.floor(slicedData1.length * 0.4)].time, position: 'belowBar', color: '#2196F3', shape: 'arrowUp', text: 'Buy' },
    { time: slicedData1[Math.floor(slicedData1.length * 0.6)].time, position: 'aboveBar', color: '#E91E63', shape: 'arrowDown', text: 'Sell' }
]);
