Bar Series Examples

Basic Bar Chart

const chart = LightweightCharts.createChart(document.getElementById('chart-container1'), {
    width: 600,
    height: 300,
});
const barSeries = chart.addBarSeries();
const randomSlice1 = getRandomSlice();
barSeries.setData(data.slice(randomSlice1.start, randomSlice1.end));

Customized Bar Chart

const chart2 = LightweightCharts.createChart(document.getElementById('chart-container2'), {
    width: 600,
    height: 300,
});
const barSeries2 = chart2.addBarSeries({
    thinBars: false,
    upColor: '#26a69a',
    downColor: '#ef5350',
});
const randomSlice2 = getRandomSlice();
barSeries2.setData(data.slice(randomSlice2.start, randomSlice2.end));
